# 설치
- sudo apt-get install ruby
- download rubygems from website
- ruby setup.rb > failed
- sudo apt-get install ruby-dev
- retry ruby setup.rb > success
- sudo gem install jekyll

# 외부에서 접속 가능하게 하기
- jekyll serve --host 0.0.0.0
